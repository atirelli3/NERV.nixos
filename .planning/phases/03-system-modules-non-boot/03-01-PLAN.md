- nerv.locale.timeZone: lib.mkOption { type = lib.types.str; default = "UTC"; description = "System time zone. See 'timedatectl list-timezones'."; example = "Europe/Rome"; }
- nerv.locale.defaultLocale: lib.mkOption { type = lib.types.str; default = "en_US.UTF-8"; description = "Default system locale for LC_* variables."; example = "it_IT.UTF-8"; }
- nerv.locale.keyMap: lib.mkOption { type = lib.types.str; default = "us"; description = "Console keymap. Run 'localectl list-keymaps' for available values."; example = "us-acentos"; }
- nerv.primaryUser: lib.mkOption { type = lib.types.listOf lib.types.str; default = []; description = "Primary system users. Each gets wheel+networkmanager groups wired automatically. If nerv.zsh.enable is true, shell is set to pkgs.zsh for each listed user."; example = [ "demon0" ]; }

config block (use lib.mkMerge for two fragments):

Fragment 1 — always-on identity wiring:

```nix
  # Terminus gives the best unicode coverage in TTY (PSF format).

}
```
