    ```
    #            The disk device is set independently via disko.devices.disk.main.device
    ```

    ```nix
        description = ''
          Disk layout type.

      };
    };
    ```

    ```nix
                    mountpoint   = "/persist";
                    mountOptions = [ "compress=zstd:3" "noatime" "space_cache=v2" ];

              };
            };
          };
    ```
